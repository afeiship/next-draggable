description:

  • Draggable simulate by touch event based on next toolkit.

usage:

  • get instance:
  var dom=nx.$('#drag1');
  var instance=new nx.ui.Draggable({
    dom:dom[0]
  });
  • dragstart:
  instance.on('dragstart',function(){
    console.log('dragstart!!!');
  });
  • drag:
  instance.on('drag',function(){
    console.log('move!!!');
  });
  • dragend:
  instance.on('dragend',function(){
    console.log('dragend!!!');
  });
